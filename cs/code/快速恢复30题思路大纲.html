                                                                                        # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseBetween(self, head: Optional[ListNode], left: int, right: int) -> Optional[ListNode]:
        """
        h：哨兵节点
        pre：上一段最后一个    begin为pre的下一个
        end：反转段的最后一个  nxt为end的下一个
        """
        def reverse_listnode(head):
            pre = None
            while head:
                nxt = head.next
                head.next = pre
                pre = head
                head = nxt
            return pre

        h = pre = ListNode(0, head)
        for _ in range(left-1):
            pre = pre.next
        begin = end = pre.next
        for _ in range(right-left):
            end = end.next
        nxt = end.next
        end.next = None
        pre.next = reverse_listnode(begin)
        begin.next = nxt
        return h.next

                                                                                        # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseKGroup(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        """
        h：哨兵节点
        pre：翻转链表上一段的最后一个  begin：待翻转链表的第一个，也是pre的下一个
        end：待翻转链表的最后一个  nxt：翻转链表下一段的第一个，也是end的下一个
        """
        h = pre = end = ListNode(0, head)
        while end:
            for _ in range(k):
                if end:
                    end = end.next
            if not end: return h.next
            nxt = end.next
            end.next = None
            begin = pre.next
            pre.next = self.reverse(begin)
            begin.next = nxt
            pre = end = begin
        return h.next
    
    def reverse(self, head: Optional[ListNode]) -> Optional[ListNode]:
        pre = None
        while head:
            nxt = head.next
            head.next = pre
            pre = head
            head = nxt
        return pre

                                                                                        # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        """
        fast：快指针，每次走2步
        slow：慢指针，每次走1步
        fast和slow都从起点出发，直到fast为null
        当相遇时，说明有环，此时fast移到起点，每次走1步，直到和slow相遇，此时，相遇节点即为环的入口点

        证明：
        a 起点到环的入口点
        b 环的入口点到相遇点
        c 相遇点到环的入口点

        fast: a + k1(b+c) + b
        slow: a + k2(b+c) + b
        2 * slow = fast
        2 (a + k2(b+c) + b) = a + k1(b+c) + b
        a + b = (k1-2k2)(b+c)
        a = (k1-2k2-1)(b + c) + c
        所以fast走a，slow走c后再走多个完整环，都到了环的入口点

        证明
        a：入环前长度 b：入环到相遇点 c：相遇点到入环点
        slow走的距离：a + k1(b+c) + b
        fast走的距离：a + k2(b+c) + b
        2(a + k1(b+c) + b) =  a + k2(b+c) + b
        2a + 2b + 2k1(b+c) = a + b + k2(b+c)
        a = (k2-2k1) (b+c) - b
        a = (k2-2k1-1)(b+c) + c
        a = k(b+c) + c
        """
        fast = slow = head
        flag = False
        res = None
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast is slow:
                flag = True
                break
        if flag:
            fast = head
            while fast is not slow:
                fast = fast.next
                slow = slow.next
            res = fast
        return res

# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reorderList(self, head: Optional[ListNode]) -> None:
        """
        Do not return anything, modify head in-place instead.
        fast快指针
        slow慢指针
        添加哨兵节点，这样会使得不论奇偶数个节点，slow都指向待翻转节点的上一个节点
        翻转后半段
        拼成一段
        """
        fast = slow = ListNode(0, head)
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
        nxt = slow.next
        slow.next = None
        second = self.reverse(nxt)
        merge = head
        while second:
            merge_nxt = merge.next
            second_nxt = second.next
            merge.next = second
            merge.next.next = merge_nxt
            merge = merge_nxt
            second = second_nxt
        
    def reverse(self, head: Optional[ListNode]) -> Optional[ListNode]: 
        pre = None
        while head:
            nxt = head.next
            head.next = pre
            pre = head
            head = nxt
        return pre

        q.append(root)
        res = list()
        while q:
            layer = list()
            n = len(q)
            for _ in range(n):
                node = q.popleft()
                if node:
                    layer.append(node.val)

# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def zigzagLevelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root: return list()
        res = list()
        q = collections.deque()
        q.append(root)
        while q:
            n = len(q)
            layer = list()
            for _ in range(n):
                node = q.popleft()
                if node:
                    layer.append(node.val)
                    if node.left: q.append(node.left)
                    if node.right: q.append(node.right)
            res.append(layer if len(res) % 2 == 0 else layer[::-1])
        return res

                                                                                        # 236 二叉树的最近公共祖先 (Done)

                                                                                        # Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
        # 只有4种情况：1）有一个就是root 2）一左一右 3）都在左边 4）都在右边
        if root in (None, p, q): # 找到p或q任意一个则不为None，1个都找不到则为None
            return root
        left = self.lowestCommonAncestor(root.left, p, q) # 左边
        right = self.lowestCommonAncestor(root.right, p, q) # 右边
        if left and right: # 两边都找到一个
            return root
        return left if left else right

                                                                                        class Solution:
    """
    时间复杂度 O(n)
    空间复杂度 O(n)
    """
    def climbStairs(self, n: int) -> int:
        if n <= 2:
            return n
        dp = [0] * (n+1)
        dp[1] = 1
        dp[2] = 2
        for i in range(3,n+1):
            dp[i] = dp[i-1] + dp[i-2]
        return dp[n]   

                                                                                        class Solution:
    """
    时间复杂度 O(n)
    空间复杂度 O(n)
    """
    def climbStairs(self, n: int) -> int:
        def climb(n, memo={}):
            if n <= 2: return n
            if n not in memo:
                memo[n] = climb(n-1, memo) + climb(n-2, memo)
            return memo[n]
        
        return climb(n)

                                                                                        # 1143 最长公共子序列 (Done)

                                                                                        class Solution:
    def longestCommonSubsequence(self, text1: str, text2: str) -> int:
        """
        思路
        1.状态定义：二维dp，dp[i][j]代表长度为i和长度为j的字符串的公共子序列长度
        2.初始状态：dp[i][0]和dp[0][j]值为0
        3.状态转移：
            如果i==j,则dp[i][j] = dp[i-1][j-1] + 1
            如果i!=j,则dp[i][j] = max(dp[i-1][j],dp[i][j-1])
        
        二维数组
        每个元素代表text1前i个字符和text2前j个字符的最长公共子序列
        状态转移
            如果i和j位置的字符相同时：dp[i][j] = dp[i-1][j-1] + 1
            不相同时：dp[i][j] = max(dp[i-1][j], dp[i][j-1])
        
        时间复杂度 O(m*n)
        空间复杂度 O(m*n)
        """
        m, n = len(text1), len(text2)
        dp = [[0]*(n+1) for _ in range(m+1)]
        for i in range(1, m+1):
            for j in range(1, n+1):
                dp[i][j] = dp[i-1][j-1] + 1 if text1[i-1] == text2[j-1] else max(dp[i-1][j],dp[i][j-1])
        return dp[m][n]

                                                                                        class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        """
        二维dp
        dp[i][j] 代表到当前位置的最小路径和
        状态转移
        dp[i][j] = grid[i][j] + min(dp[i-1][j],dp[i][j-1])

        时间复杂度 O(m*n)
        空间复杂度 O(m*n)
        """
        m, n = len(grid), len(grid[0])
        dp = [[0]*n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    dp[i][j] = grid[i][j]
                elif i == 0:
                    dp[i][j] = grid[i][j] + dp[i][j-1]
                elif j == 0:
                    dp[i][j] = grid[i][j] + dp[i-1][j]
                else:
                    dp[i][j] = grid[i][j] + min(dp[i-1][j],dp[i][j-1])
        return dp[m-1][n-1]

                                                                                        # 5 最长回文子串 (Done)

                                                                                        class Solution:
    def longestPalindrome(self, s: str) -> str:
        """
        二维dp
        dp[i][j] 代表从下标i到j的子串是否是回文子串，如果是值为1，否则为0
        形状：上三角
        填充：从中间往右上角填充（size从小到大）
        状态转移(只记录回文串)

        时间复杂度 O(n^2)
        空间复杂度 O(n^2)
        """
        n = len(s)
        dp = [[0]*n for _ in range(n)]
        res_start = 0
        res_length = 1
        for length in range(1,n+1):
            for start in range(n-length+1):
                if length == 1:
                    dp[start][start+length-1] = 1
                elif length == 2:
                    dp[start][start+length-1] = s[start] == s[start+length-1]
                else:
                    dp[start][start+length-1] = s[start] == s[start+length-1] and dp[start+1][start+length-2]
                if dp[start][start+length-1] == 1 and res_length < length: 
                    res_start = start
                    res_length = length
        return s[res_start:res_start+res_length]

                                                                                        # 300 最长上升子序列 (Done) Note:没法用单调栈

                                                                                        class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        """
        一维dp
        dp[i]代表以下标i结尾的最长递增子序列
        dp[i]由比i小的下标dp中，比nums[i]小的里面最大的dp值

        时间复杂度 O(n^2)
        空间复杂度 O(n)
        """
        dp = [1] * len(nums)
        res = 1
        for i in range(len(nums)):
            maxseqlen = 0
            for j in range(i):
                if dp[j] > maxseqlen and nums[j] < nums[i]:
                    maxseqlen = dp[j]
            dp[i] = maxseqlen + 1
            res = max(res, dp[i])
        return res

                                                                                        class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def backtrack(index, arr): # 回溯
            if index == len(nums): # 回溯结束条件
                # ⚠️这里要注意，要重新复制一份，不在原地址上进行变动
                res.append(arr[:]) # 将当前排列添加到结果中
            else:
                for num in nums: # 每个下标，遍历每一个数字
                    if num in use: # 如果已经使用过，则跳过
                        continue
                    use.add(num) # 将本次使用的num添加到use中
                    arr.append(num) # 将本次使用的num接到arr后面
                    backtrack(index + 1, arr) # 进行下一个下标的处理
                    arr.pop() # 恢复arr
                    use.remove(num) # 恢复use

        res = list() # 最终结果
        use = set() # 已使用的数字
        backtrack(0, list()) # 从第1个下标的位置看起，arr初始化为空列表
        return res

                res1 = ["".join(a) for a in res1]
                res.append(res1)
            else:
                for i in range(n):
                    if i in use or (index - i) in use1 or (index + i) in use2:
                        continue
                    use.add(i)
                    use1.add(index - i)
                    use2.add(index + i)
                    arr.append(i)
                    backtrack(index+1,arr)
                    arr.pop()
                    use.remove(i)

        use1 = set()
        use2 = set()
        backtrack(0,list())
        return res

                                                                                        class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        def dfs(row, col):
            grid[row][col] = "0"
            for x, y in [(row-1,col),(row+1,col),(row,col-1),(row,col+1)]:
                if 0<=x<m and 0<=y<n and grid[x][y] == "1":
                    dfs(x,y)
        
        def bfs(row,col):
            q = deque()
            q.append((row,col))
            grid[row][col] = "0"
            while q:
                x,y = q.popleft()
                for x1, y1 in [(x-1,y),(x+1,y),(x,y-1),(x,y+1)]:
                    if 0<=x1<m and 0<=y1<n and grid[x1][y1] == "1":
                        q.append((x1,y1))
                        grid[x1][y1] = "0"
                
        m, n = len(grid), len(grid[0])
        res = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == "1":
                    res += 1
                    bfs(i,j) # dfs(i,j)
        return res

                                                                                        class Solution:
    def nearestExit(self, maze: List[List[str]], entrance: List[int]) -> int:
        m, n = len(maze), len(maze[0])
        q = deque()
        q.append(entrance)
        maze[entrance[0]][entrance[1]] = "+"
        res = 0
        while q:
            res += 1
            nums = len(q)
            # print(q)
            for _ in range(nums):
                x,y = q.popleft()
                for x1,y1 in [(x-1,y),(x+1,y),(x,y-1),(x,y+1)]:
                    if 0<=x1<m and 0<=y1<n and maze[x1][y1] == ".":
                        if x1 == 0 or x1 == m-1 or y1 == 0 or y1 == n-1:
                            return res

                                                                                        class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        """
        单调队列（递减队列，保证在队列里的第一个始终是最大的，这样往右移动一位后，只需比较队列头的元素即可）
        1.定义单调队列q，队列里面存下标，不存数
        2.遍历数组，更新单调队列q
        3.单调队列头部元素对应值为最大值
        """
        q = deque()
        res = list()
        for i in range(len(nums)):
            # 更新q
            while q and nums[q[-1]] <= nums[i]:
                q.pop()
            q.append(i)
            if q[0] <= i - k:
                q.popleft()
            # 队列头部元素对应值为最大值
            if i >= k - 1:
                res.append(nums[q[0]])
        return res


                                                                                        class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        """
        思路大纲
        大顶堆，通过将小顶堆取负数来表示
        每个元素是一个元组，第一列为负数，第二列为下标
        取堆顶作为最大值
        通过下标来断定堆顶下标是否超出范围，如果超出则去掉
        """
        heap = [(-nums[i], i) for i in range(k-1)]
        heapq.heapify(heap)
        res = list()
        for i in range(k-1, len(nums)):
            heapq.heappush(heap, (-nums[i], i))
            while heap[0][1] <= i - k:
                heapq.heappop(heap)
            res.append(-heap[0][0])
        return res

            prev_sign = "+" # 运算符，不立即计算，先存起来
            while len(s) > 0:

                # 遇到左括号开始递归计算 num
                elif c == '(': # 判断当前字符是左括号
                    num = dfs(s) # 递归

                    return sum(nums) # 遇到右括号返回递归结果

                    num = 0
                    prev_sign = c
            return sum(nums)
        
        s = s.replace(" ", "")
        s = s.replace(')', '+)')+'+' # 右括号前都多了一个➕，在最后也添加了一个➕ (此行是避免特判右括号前和最后一位的处理)
        return dfs(collections.deque(s))

        """

    def __init__(self, key=0, value=0):
        self.key = key
        self.value = value

class LRUCache:
    """
    解题关键：
    1.双向链表（链表头部表示最近使用的，链表尾部表示最近未使用的，可以支持在链表头尾进行操作）
    2.哈希表, k是int类型， v是LinkedNode类型（v必须是LinkedList类型，因为需要通过k立即找到这个node，对node进行移动到头部操作）
    """
    def __init__(self, capacity: int):
        self.cache = dict() # 哈希表, k是int类型， v是LinkedNode类型
        self.head = LinkedNode() # 头部哨兵节点
        self.tail = LinkedNode() # 尾部哨兵节点
        # 构造双向链表
        self.head.next = self.tail 
        self.tail.next = self.head
        # 容量 和 使用大小
        self.capacity = capacity
        self.size = 0

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1
        node = self.cache[key]
        self.moveToHead(node)
        return node.value

    def put(self, key: int, value: int) -> None:
        if key not in self.cache:
            node = LinkedNode(key, value)
            self.cache[key] = node
            self.addToHead(node)
            self.size += 1
            if self.size > self.capacity:
                removed = self.removeTail()
                self.cache.pop(removed.key)
                self.size -= 1
        else:
            node = self.cache[key]
            node.value = value
            self.moveToHead(node)

    def addToHead(self, node):
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node

    def removeNode(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def moveToHead(self, node):
        self.removeNode(node)
        self.addToHead(node)

    def removeTail(self):
        node = self.tail.prev
        self.removeNode(node)
        return node

                                                                                        class RandomizedCollection:
        """
        思路大纲
        1.定义一个dict和list。dict的k是值，v是值在list中存放的下标。
        2.删除时，把list的最后一位移到删除的值的位置，来确保O(1)
        """

    def __init__(self):
        self.mydict = defaultdict(set)
        self.values = list()

    def insert(self, val: int) -> bool:
        self.values.append(val)
        self.mydict[val].add(len(self.values)-1)
        return len(self.mydict[val]) == 1

    def remove(self, val: int) -> bool:
        if val in self.mydict:
            last_index = len(self.values) - 1
            change_index = self.mydict[val].pop()
            self.values[change_index] = self.values[last_index]
            self.values.pop()
            if last_index != change_index:
                self.mydict[self.values[change_index]].remove(last_index)
                self.mydict[self.values[change_index]].add(change_index)
            if len(self.mydict[val]) == 0:
                del self.mydict[val]
            return True
        else:
            return False

    def getRandom(self) -> int:
        return random.choice(self.values)

    def sortArray(self, nums: List[int]) -> List[int]:
        def merge_sort(left, right):
            if left >= right: return
            mid = (left + right) // 2
            merge_sort(left, mid)
            merge_sort(mid + 1, right)
            merge(left, mid, mid + 1, right)

            nonlocal nums
            tmp = nums[l1:r2+1] # ⚠️这里不能是tmp = nums，否则会超时
            start = cur = l1
            while l1 <= r1 and l2 <= r2:
                if tmp[l1-start] <= tmp[l2-start]:
                    nums[cur] = tmp[l1-start]
                    cur += 1
                    l1 += 1
                else:
                    nums[cur] = tmp[l2-start]
                    cur += 1
                    l2 += 1
            while l1 <= r1:
                nums[cur] = tmp[l1-start]
                cur += 1
                l1 += 1
            while l2 <= r2:
                nums[cur] = tmp[l2-start]
                cur += 1
                l2 += 1

        merge_sort(0, len(nums) - 1)

        return nums
